programa dudamelhorado

// Listas oficiais
variavel listaAlunos : vetor[100] de cadeia
variavel listaProfessores : vetor[50] de cadeia

// Fila de entrada (pode conter alunos e professores misturados)
variavel filaEntrada : vetor[200] de cadeia
variavel i : inteiro

inicio
    // Enquanto houver pessoas na fila
    para i de 1 ate tamanho(filaEntrada) faca
        se filaEntrada[i] está em listaAlunos então
            escreva("Entrada permitida para o ALUNO: ", filaEntrada[i])
        senao se filaEntrada[i] está em listaProfessores então
            escreva("Entrada permitida para o PROFESSOR: ", filaEntrada[i])
        senao
            escreva("ERRO: ", filaEntrada[i], " não consta em nenhuma lista oficial. Entrada negada.")
        fimse
    fimpara

    escreva("Verificação concluída. Todos da fila foram processados.")
fimalgoritmo
